Dynamic Avatar Widget Integration

Overview

This example demonstrates how to integrate the AI Avatar Widget with fully customizable parameters. Users can easily change the knowledge base, avatar, and opening message without editing the core code.

Step 1: Add CSS and Widget Container

<!-- Include widget styles -->
<link rel="stylesheet" href="https://your-cdn.com/assets/ai-avatar.css">

<!-- Widget container with data attributes -->
<div id="ai-chat-widget" 
     data-knowledge-id="your_knowledge_id" 
     data-avatar-name="Tyler-insuit-20220721" 
     data-opening-message="Hello! I'm Aiva. How can I help you today?"
     style="position: fixed; bottom: 20px; right: 20px; z-index: 9999;"></div>

Place the widget container where you want the chat widget to appear. The position can be adjusted using the style attribute.

Step 2: Add Script Dependencies

<!-- Load React dependencies -->
<script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
<script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

<!-- Load widget script -->
<script src="https://your-cdn.com/assets/avatar-widget.umd.js"></script>

Step 3: Initialize Widget

<script>
  document.addEventListener('DOMContentLoaded', function() {
    try {
      // Get the container element
      const container = document.getElementById('ai-chat-widget');
      
      // Check if widget function exists
      if (typeof window.renderAvatarWidget !== 'function') {
        console.error('Avatar widget function not found!');
        return;
      }
      
      // Get configuration from data attributes
      const config = {
        knowledgeID: container.getAttribute('data-knowledge-id'),
        avatarId: container.getAttribute('data-avatar-name'),
        openingMessage: container.getAttribute('data-opening-message')
      };
      
      // Initialize the widget
      window.renderAvatarWidget('ai-chat-widget', config);
      
    } catch (error) {
      console.error('Error initializing widget:', error);
    }
  });
</script>

Step 4: One-Line Integration (Alternative)

For a simplified integration, use this all-in-one snippet:

<!-- AI Avatar Widget - All-in-one Integration -->
<div id="ai-chat-widget" data-knowledge-id="your_knowledge_id"></div>

<script>
  (function(d,s,id){
    // Create container if not exists
    var c = d.getElementById(id) || d.createElement('div');
    if(!d.getElementById(id)) {
      c.id = id;
      c.style = "position:fixed;bottom:20px;right:20px;z-index:9999;";
      d.body.appendChild(c);
    }
    
    // Add stylesheet
    var l = d.createElement('link');
    l.rel = 'stylesheet';
    l.href = 'https://your-cdn.com/assets/ai-avatar.css';
    d.head.appendChild(l);
    
    // Add React dependencies
    function loadScript(src, callback) {
      var js = d.createElement('script');
      js.src = src;
      js.crossOrigin = 'anonymous';
      js.onload = callback;
      d.head.appendChild(js);
    }
    
    function initWidget() {
      // Load widget script
      var ws = d.createElement('script');
      ws.src = 'https://your-cdn.com/assets/avatar-widget.umd.js';
      ws.onload = function() {
        // Initialize widget with data attributes
        if(typeof window.renderAvatarWidget === 'function') {
          window.renderAvatarWidget(id, {
            knowledgeID: c.getAttribute('data-knowledge-id'),
            avatarId: c.getAttribute('data-avatar-name') || 'Tyler-insuit-20220721',
            openingMessage: c.getAttribute('data-opening-message') || 'Hello! I\'m Aiva. How can I help you today?'
          });
        }
      };
      d.head.appendChild(ws);
    }
    
    // Check if React is already loaded
    if(typeof React === 'undefined') {
      loadScript('https://unpkg.com/react@18/umd/react.production.min.js', function() {
        loadScript('https://unpkg.com/react-dom@18/umd/react-dom.production.min.js', initWidget);
      });
    } else if(typeof ReactDOM === 'undefined') {
      loadScript('https://unpkg.com/react-dom@18/umd/react-dom.production.min.js', initWidget);
    } else {
      initWidget();
    }
  })(document, 'script', 'ai-chat-widget');
</script>

Configuration Parameters

Parameter Data Attribute Description Status
knowledgeID data-knowledge-id Your unique knowledge base identifier Required
avatarId data-avatar-name The avatar to use in the chat widget Optional
openingMessage data-opening-message The first message shown to users Optional
theme data-theme Color theme (light or dark) Optional
position data-position Widget position (bottom-right, bottom-left, etc.) Optional
